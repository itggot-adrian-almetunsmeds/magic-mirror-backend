def self.insert(db, position, value)
  positions = position[0]
  position.shift
  if position.length >= 1
    position.each do |z|
      positions += ', ' + z
    end
  end

  values = '?'
  (value.length - 1).times { |_| values += ', ?' }
  @db.execute("INSERT INTO #{db} (#{positions}) VALUES (#{values})", value[0..value.length])
end
